214 /**
215  * Coordinates vector
216  * @param {number} x Horizontal component.
217  * @param {number} y Vertical component.
218  * @return {Coords} Vector coordinates.
219  * @constructor
220  * @struct
221  */
222 function Coords(x, y) {
223 
224     /**
225      * Horizontal component.
226      * @expose
227      * @type {number}
228      */
229     this.x = x;
230 
231     /**
232      * Vertical component.
233      * @expose
234      * @type {number}
235      */
236     this.y = y;
237 
238     /**
239      * Returns the coords plus an other coords c
240      * @param {Coords} c Other vector's coordinates.
241      * @this {Coords}
242      * @return {Coords} New vector's coordinates.
243      */
244     this.plus = function(c) {
245         return coords(this.x + c.x, this.y + c.y);
246     };
247 
248     /**
249      * Returns the coords minus an other coords c
250      * @param {Coords} c Other vector's coordinates.
251      * @this {Coords}
252      * @return {Coords} New vector's coordinates.
253      */
254     this.minus = function(c) {
255         return coords(this.x - c.x, this.y - c.y);
256     };
257 
258     /**
259      * Multiplies the coords by a scalar
260      * @param {number} k Multiplication factor.
261      * @this {Coords}
262      * @return {Coords} New vector's coordinates.
263      */
264     this.times = function(k) {
265         return coords(this.x * k, this.y * k);
266     };
267 
268     /**
269      * Returns the scalar products of the coords with an other coords c
270      * @param {Coords} c Other vector's coordinates.
271      * @this {Coords}
272      * @return {number} Product value.
273      */
274     this.scalar = function(c) {
275         return this.x * c.x + this.y * c.y;
276     };
277 
278     /**
279      * Returns the coords norm, i.e. vector modulus.
280      * @this {Coords}
281      * @return {number} Vector norm.
282      */
283     this.norm = function() {
284         return Math.sqrt(this.scalar(this));
285     };
286 
287     /**
288      * Returns the distance between this and c.
289      * @param {Coords} c The second coords.
290      * @this {Coords}
291      * @return {number} The distance.
292      */
293     this.distance = function(c) {
294         return this.minus(c).norm();
295     };
296 
297     return this;
298 }

300 /**
301  * @param {number} x Horizontal component.
302  * @param {number} y Vertical component.
303  * @return {Coords} Vector coordinates.
304  */
305 function coords(x, y) {
306     return new Coords(x, y);
307 }

309 /**
310  * Graph vertex (also called node)
311  * @param {Object} props Vertex properties (the same object w/o methods).
312  * @return {Vertex} Vertex object.
313  * @constructor
314  * @struct
315  */
316 function Vertex(props) {
317 
318     if (!props.members || !(props.members instanceof Array)) props.members = [];
319 
320     if (!props.position)
321         props.position = coords((Math.random() - .5) * context.width,
322                                 (Math.random() - .5) * context.height);
323 
324     if (!props.color || typeof props.color !== 'string')
325     {
326         /** @type {integer} */
327         var nColor = Math.round(Math.random() * predefinedColors.length);
328 
329         props.color = predefinedColors[nColor];
330     }
331 
332     /** @type {string} */
333     this.title = props.title;
334 
335     /** @type {string} */
336     this.type = props.type;
337 
338     /** @type {{src: string, width: number, height: number}} */
339     this.image = props.image;
340 
341     /** @type {string} */
342     this.description = props.description;
343 
344     /** @type {Object.<string, string>} */
345     this.details = props.details;
346 
347     /** @type {number} */
348     this.mass = minimalMass + props.members.length;
349 
350     /** @type {number} */
351     this.radius = Math.sqrt(this.mass) * objectsDensity;
352 
353     /** @type {Coords} */
354     this.position = props.position;
355 
356     /** @type {Object.<string, string>} */
357     this.links = props.links;
358 
359     /**
360      * @expose
361      * @type {Array.<number, Object>}
362      */
363     this.members = props.members;
364 
365     /** @type {string} */
366     this.color = props.color;
367 
368     /**
369      * Returns the distance from this vertex to the vertex n
370      * @param {Vertex} v The other vertex.
371      * @this {Coords}
372      * @return {number} The distance ( 0).
373      */
374     this.distance = function(v) {
375         return this.position.distance(v.position);
376     };
377 
378     /**
379      * Computes the speed of the node
380      *
381      * The node speed is determined by two attraction trap:
382      * * Each node attracts other nodes of the same level,
383      * * Each node is attracted by the center of the parent node.
384      *
385      * @this {Vertex}
386      * @return {Coords} The speed vector.
387      */
388     this.speed = function() {
389 
390         /** @type {Coords} */
391         var s = coords(0, 0);
392 
393         for (var v = 0, w = graph.vertices.length; v < w; v++)
394         {
395             /** @type {Vertex} */
396             var n = vertex(graph.vertices[v]);
397 
398             if (n.title != this.title)
399             {
400                 var x = n.distance(this) /
401                         (this.radius + n.radius + objectsMargin);
402 
403                 /** @type {number} */
404                 var f = (x - 1) / (x * x);
405 
406                 s = s.plus(n.position.minus(this.position)
407                     .times(f / n.distance(this)));
408             }
409         }
410 
411         s = s.plus(this.position.times(-.002 * this.mass));
412 
413         return s;
414     };
415 
416     /**
417      * Updates the node's position after a time step, depending on their.
418      * We use for this a simple <a href="//en.wikipedia.org/wiki/Euler_method">
419      * Euler's method</a>.
420      * @this {Vertex}
421      * @return {Vertex} New vertex.
422      */
423     this.step = function() {
424 
425         this.position = this.position.plus(this.speed().times(timeStep));
426 
427         return this;
428     };
429 
430     /**
431      * Draw the vertex
432      * @param {boolean} selected Whether the vertex is selected.
433      * @this {Vertex}
434      * @return {Vertex} The same vertex.
435      */
436     this.draw = function(selected) {
437 
438         /** @type {Coords} */
439         var position = this.position.plus(context.center);
440 
441         /** @type {number} */
442         var opacity = selected ? 1 : .6;
443 
444         /** @type {number} */
445         var radius = this.radius;
446 
447         /** @type {integer} */
448         var textSize = Math.ceil(2.5 * Math.sqrt(radius));
449 
450         context.beginPath();
451         context.lineWidth = selected ? 3 : 1;
452         context.fillStyle = 'rgba(' + this.color + ',' + (opacity / 2) + ')';
453         context.strokeStyle = 'rgba(' + this.color + ',' + (opacity) + ')';
454 
455         // Circle for organizations
456         if (this.type == vertexTypes[0])
457             // Optimization : Math.PI * 2 = 6.283185
458             context.arc(position.x, position.y, radius, 0, 6.283185);
459 
460         // Square for people
461         else if (this.type == vertexTypes[1])
462             this.path([position.plus(coords(- radius, - radius)),
463                         position.plus(coords(radius, - radius)),
464                         position.plus(coords(radius, radius)),
465                         position.plus(coords(- radius, radius))]);
466 
467         // Diamond for projects (and others if any)
468         else this.path([position.plus(coords(- radius, 0)),
469                         position.plus(coords(0, - radius)),
470                         position.plus(coords(radius, 0)),
471                         position.plus(coords(0, radius))]);
472 
473         context.closePath();
474         context.stroke();
475         context.fill();
476 
477         context.fillStyle = '#000';
478         context.font = '600 ' + textSize + 'px Helvetica';
479         context.textAlign = 'center';
480         context.fillText(this.title, position.x, position.y + textSize / 4);
481 
482         return this;
483     };
484 
485     /**
486      * @param {Vertex} v Targeted vertex.
487      * @param {boolean} selected Whether the vertex is selected.
488      * @this {Vertex}
489      * @return {Vertex} The same vertex.
490      */
491     this.drawLinkTo = function(v, selected) {
492 
493         /** @type {Coords} */
494         var position = this.position.plus(context.center);
495 
496         /** @type {Coords} */
497         var target = v.position.plus(context.center);
498 
499         /** @type {Coords} */
500         var axis = target.minus(position);
501 
502         /** @type {number} */
503         var distance = position.distance(target);
504 
505         /** @type {Coords} */
506         var arrowStart = axis.times(this.radius / distance).plus(position);
507 
508         /** @type {Coords} */
509         var arrowEnd = axis.times(-1 * v.radius / distance).plus(target);
510 
511         /** @type {Coords} */
512         var arrowWing1 = coords(axis.x - axis.y, axis.y + axis.x)
513                         .times(-.5 * 15 / distance).plus(arrowEnd);
514 
515         /** @type {Coords} */
516         var arrowWing2 = coords(axis.x + axis.y, axis.y - axis.x)
517                         .times(-.5 * 15 / distance).plus(arrowEnd);
518 
519         context.beginPath();
520         context.lineWidth = selected ? 3 : 1;
521         context.strokeStyle = 'rgba(' + this.color + ',.8)';
522 
523         this.path([arrowStart, arrowEnd, arrowWing1])
524             .path([arrowEnd, arrowWing2]);
525 
526         context.stroke();
527 
528         return this;
529     };
530 
531     /**
532      * @param {Array.<Coords>} coordsList Coords list of path to draw.
533      * @this {Vertex}
534      * @return {Vertex} The same vertex.
535      */
536     this.path = function(coordsList) {
537 
538         for (var c = 0, l = coordsList.length; c < l; c++)
539 
540             if (c == 0) context.moveTo(coordsList[c].x, coordsList[c].y);
541             else context.lineTo(coordsList[c].x, coordsList[c].y);
542 
543         return this;
544     };
545 
546     return this;
547 }

557 /**
558  * Search the graph for a vertex called "title" and return its index
559  * @param {string} title The vertex title.
560  * @return {integer} the index of the found node in the graph array.
561  */
562 function getVertexByTitle(title) {
563 
564     for (var v = 0, w = graph.vertices.length; v < w; v++)
565         if (graph.vertices[v].title == title) return v;
566 
567     return -1;
568 }

570 /**
571  * Animates the graph
572  */
573 function animate() {
574 
575     // First of all, we erase everything on the canvas.
576     context.clearRect(0, 0, context.width, context.height);
577 
578     // Then we (re)draw the graph.
579     for (var i = 0, j = graph.vertices.length; i < j; i++)
580     {
581         var selected = (selectedVertices.indexOf(i) !== -1);
582 
583         graph.vertices[i] = vertex(graph.vertices[i]).step().draw(selected);
584 
585         // We also draw also to symbolize links between vertices.
586         for (var l in graph.vertices[i].links)
587             graph.vertices[i].drawLinkTo(
588                 vertex(graph.vertices[getVertexByTitle(l)]), selected);
589     }
590 
591     setTimeout(animate, timeStep);
592 }

594 /**
595  * Adjust the canvas size to the window
596  */
597 function setCanvasSize() {
598 
599     /** @type {number} */
600     var height = window.innerHeight;
601 
602     /** @type {number} */
603     var width = window.innerWidth;
604 
605     if (context.width != width || context.height != height) {
606 
607         /** @type {Node} */
608         var canvas = getId('canvas');
609         canvas.width = width;
610         canvas.height = height;
611         context.width = width;
612         context.height = height;
613         context.center = coords(width / 2, height / 2);
614     }
615 
616     // We resize canvas every seconds in case of window's resizing
617     window.setTimeout(setCanvasSize, 1000);
618 }

620 /**
621  * Search the graph for the closest vertex from the coords "c"
622  * @param {Coords} c The origin vector.
623  * @return {integer} The closest vertex index in the graph array.
624  */
625 function getClosestTo(c) {
626 
627     /** @type {integer} */
628     var closest;
629 
630     /** @type {number} */
631     var dist = Infinity;
632 
633     /** @type {number} */
634     var ndist;
635 
636     for (var v = 0, w = graph.vertices.length; v < w; v++)
637     {
638         /** @type {Vertex} */
639         var nvertex = vertex(graph.vertices[v]);
640         ndist = c.minus(nvertex.position).norm();
641 
642         if (ndist < dist)
643         {
644             dist = ndist;
645             closest = v;
646         }
647     }
648 
649     return closest;
650 }

652 /**
653  * @param {Coords} c The coords searched.
654  * @return {integer} The vertex index, if any, located here.
655  */
656 function getVertexByCoords(c) {
657 
658     for (var v = 0, nv = graph.vertices.length; v < nv; v++)
659     {
660         /** @type {Vertex} */
661         var w = vertex(graph.vertices[v]);
662 
663         if (c.minus(w.position).norm() < w.radius) return v;
664     }
665 
666     return -1;
667 }

779 /**
780  * Creates a new vertex and opens the Edit panel
781  */
782 function addNewVertex() {
783 
784     /** @type {integer} */
785     var newVertexTitle = 1;
786 
787     while (getVertexByTitle('#' + newVertexTitle) != -1)
788     {
789         if (newVertexTitle > 99) return;
790         else newVertexTitle++;
791     }
792 
793     graph.vertices.push(vertex({title: '#' + newVertexTitle,
794                             type: vertexTypes[0],
795                             position: context.mouse}));
796 
797     selectedVertices = [graph.vertices.length - 1];
798 
799     editPanel();
800 }
